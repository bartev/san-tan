\begin{itemize}
  \item Set a tab stop with \verb!\=!
  \item End a row with \verb!\\!
  \item Go to the next tab stop with \verb!\>!
\end{itemize}

\begin{verbatim}
\begin{tabbing}
\emph{Info:}  \= Software \= : \= \LaTeX \\
\> Author \> : \> Leslie Lamport \\
\> Website \> : \> www.latex-project.org
\end{tabbing}
\end{verbatim}
